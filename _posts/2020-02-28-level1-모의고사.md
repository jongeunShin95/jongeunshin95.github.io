## 모의고사

```python
# 제일 큰 수 빼고 0으로 만들기
def rank(temp_answer, i, j):
    if temp_answer[i] < temp_answer[j]: temp_answer[i] = 0
    elif temp_answer[i] > temp_answer[j]:   temp_answer[j] = 0
    
def solution(answers):
    answer = []
    temp_answer = [0, 0, 0]
    
    man1 = [1, 2, 3, 4, 5]
    man2 = [2, 1, 2, 3, 2, 4, 2, 5]
    man3 = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    
    # 정답 횟수 기록
    for i in range(0, len(answers)):
        if man1[i%5] == answers[i]: temp_answer[0] += 1
        if man2[i%8] == answers[i]: temp_answer[1] += 1
        if man3[i%10] == answers[i]:    temp_answer[2] += 1
    
    if temp_answer == [0, 0, 0]:    return []
    
    rank(temp_answer, 0, 1)
    rank(temp_answer, 0, 2)
    rank(temp_answer, 1, 2)
    
    # 0이 아닌 경우 해당 인덱스 번호를 배열에 추가
    for i in range(0, 3):
        if temp_answer[i] != 0: answer.append(i+1)
    
    return answer
```
